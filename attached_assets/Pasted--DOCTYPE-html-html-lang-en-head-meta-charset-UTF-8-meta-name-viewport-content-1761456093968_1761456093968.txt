    <script>
        // Initialize map
        let map = L.map('map').setView([41.0082, 28.9784], 13);
        let routingControl = null;
        let markers = [];

        // Add OpenStreetMap tile layer
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            attribution: 'Â© OpenStreetMap contributors',
            maxZoom: 19
        }).addTo(map);

        // Utility function to parse coordinates
        function parseCoordinates(input) {
            if (!input || !input.trim()) return null;
            
            const cleaned = input.trim().replace(/\s+/g, '');
            const parts = cleaned.split(',');
            
            if (parts.length !== 2) return null;
            
            const lat = parseFloat(parts[0]);
            const lng = parseFloat(parts[1]);
            
            if (isNaN(lat) || isNaN(lng)) return null;
            if (lat < -90 || lat > 90 || lng < -180 || lng > 180) return null;
            
            return { lat, lng };
        }

        // Create custom marker
        function createCustomMarker(lat, lng, type, label) {
            let className = 'marker-' + type;
            let icon = L.divIcon({
                className: 'custom-marker ' + className,
                html: label || '',
                iconSize: [32, 32],
                iconAnchor: [16, 16]
            });
            
            return L.marker([lat, lng], { icon: icon });
        }

        // Show error message
        function showError(message) {
            const errorEl = document.getElementById('error');
            errorEl.textContent = message;
            errorEl.classList.add('visible');
            setTimeout(() => errorEl.classList.remove('visible'), 5000);
        }

        // Show loading state
        function setLoading(isLoading) {
            document.getElementById('loading').classList.toggle('visible', isLoading);
        }

        // Update route info
        function updateRouteInfo(distance, time, waypointCount) {
            document.getElementById('totalDistance').textContent = distance;
            document.getElementById('estimatedTime').textContent = time;
            document.getElementById('waypointCount').textContent = waypointCount;
            document.getElementById('routeInfo').classList.add('visible');
        }

        // Clear previous route and markers
        function clearMap() {
            if (routingControl) {
                map.removeControl(routingControl);
                routingControl = null;
            }
            markers.forEach(marker => map.removeLayer(marker));
            markers = [];
        }

        // Main function to display route
        function displayRoute() {
            // Clear previous state
            clearMap();
            document.getElementById('error').classList.remove('visible');
            document.getElementById('routeInfo').classList.remove('visible');

            // Parse pickup location
            const pickupInput = document.getElementById('pickup').value;
            const pickup = parseCoordinates(pickupInput);
            
            if (!pickup) {
                showError('Invalid pickup location. Use format: latitude,longitude (e.g., 41.0082,28.9784)');
                return;
            }

            // Parse destination
            const destinationInput = document.getElementById('destination').value;
            const destination = parseCoordinates(destinationInput);
            
            if (!destination) {
                showError('Invalid destination. Use format: latitude,longitude');
                return;
            }

            // Parse via points (filter out empty ones)
            const viaPoints = [];
            for (let i = 1; i <= 5; i++) {
                const viaInput = document.getElementById(`via${i}`).value;
                if (viaInput && viaInput.trim()) {
                    const viaCoord = parseCoordinates(viaInput);
                    if (viaCoord) {
                        viaPoints.push(viaCoord);
                    } else {
                        showError(`Invalid via point ${i}. Skipping it.`);
                    }
                }
            }

            // Build waypoints array for routing
            const waypoints = [
                L.latLng(pickup.lat, pickup.lng),
                ...viaPoints.map(via => L.latLng(via.lat, via.lng)),
                L.latLng(destination.lat, destination.lng)
            ];

            setLoading(true);

            // Create routing control
            try {
                routingControl = L.Routing.control({
                    waypoints: waypoints,
                    routeWhileDragging: false,
                    addWaypoints: false,
                    draggableWaypoints: false,
                    fitSelectedRoutes: true,
                    showAlternatives: false,
                    lineOptions: {
                        styles: [{ color: '#667eea', weight: 5, opacity: 0.7 }]
                    },
                    createMarker: function(i, waypoint, n) {
                        // Don't create default markers, we'll create custom ones
                        return null;
                    }
                }).addTo(map);

                // Add custom markers
                const pickupMarker = createCustomMarker(pickup.lat, pickup.lng, 'pickup', 'P');
                pickupMarker.bindPopup('<b>Pickup Location</b>');
                pickupMarker.addTo(map);
                markers.push(pickupMarker);

                viaPoints.forEach((via, index) => {
                    const viaMarker = createCustomMarker(via.lat, via.lng, 'via', (index + 1).toString());
                    viaMarker.bindPopup(`<b>Via Point ${index + 1}</b>`);
                    viaMarker.addTo(map);
                    markers.push(viaMarker);
                });

                const destMarker = createCustomMarker(destination.lat, destination.lng, 'destination', 'D');
                destMarker.bindPopup('<b>Destination</b>');
                destMarker.addTo(map);
                markers.push(destMarker);

                // Handle route found event
                routingControl.on('routesfound', function(e) {
                    setLoading(false);
                    const routes = e.routes;
                    const summary = routes[0].summary;
                    
                    const distance = (summary.totalDistance / 1000).toFixed(2) + ' km';
                    const time = Math.round(summary.totalTime / 60) + ' min';
                    const waypointCount = viaPoints.length + 2; // pickup + vias + destination
                    
                    updateRouteInfo(distance, time, waypointCount);
                });

                // Handle routing error
                routingControl.on('routingerror', function(e) {
                    setLoading(false);
                    showError('Could not calculate route. Please check your coordinates and try again.');
                });

            } catch (error) {
                setLoading(false);
                showError('Error creating route: ' + error.message);
                console.error('Routing error:', error);
            }
        }

        // Demo data loader (optional - for testing)
        function loadDemoData() {
            document.getElementById('pickup').value = '41.0082,28.9784';
            document.getElementById('via1').value = '41.0120,28.9800';
            document.getElementById('via2').value = '41.0180,28.9850';
            document.getElementById('destination').value = '41.0250,28.9950';
        }

        // Allow Enter key to trigger route display
        document.querySelectorAll('input').forEach(input => {
            input.addEventListener('keypress', function(e) {
                if (e.key === 'Enter') {
                    displayRoute();
                }
            });
        });

        // loadDemoData();
